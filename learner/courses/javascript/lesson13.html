Lesson 13: Error Handling

Master the art of handling errors gracefully to build robust, reliable applications.

Understanding Errors

Errors are inevitable in programming. They can occur due to invalid user input, network failures, programming mistakes, or external service issues. Good error handling prevents crashes, provides helpful feedback to users, and makes debugging easier. In JavaScript, errors are objects that contain information about what went wrong.

Key Concept: Error handling is not just about preventing crashes—it's about creating a good user experience and making your code maintainable. Anticipate errors, handle them gracefully, and provide clear feedback.

Error Types

JavaScript has several built-in error types:

const error = new Error("Something went wrong");

class ValidationError extends Error {
    constructor(message, field) {
        super(message);
        this.name = "ValidationError";
        this.field = field;
    }
}

Built-in Error Types:
  • Error - Generic error, base for all others
  • SyntaxError - Code parsing error (usually during development)
  • ReferenceError - Accessing undefined variable
  • TypeError - Operation on wrong type
  • RangeError - Value outside valid range

try...catch...finally

The fundamental error handling structure:

// Basic try...catch
try {
    // Code that might throw an error
    const result = JSON.parse('invalid json');
} catch (error) {
    // Handle the error
    console.log("Caught error:", error.message);
}

// try...catch...finally
try {
    const data = riskyOperation();
    processData(data);
} catch (error) {
    console.log("Error:", error.message);
    notifyUser("Operation failed");
} finally {
    // Always runs, regardless of success or error
    cleanup();
}

// Catch block only catches errors in the try block
try {
    const result = divide(10, 0);
    console.log(result);
} catch (error) {
    console.log("Division failed:", error.message);
}

function divide(a, b) {
    if (b === 0) {
        throw new Error("Cannot divide by zero");
    }
    return a / b;
}

// Nested try...catch
try {
    try {
        throw new Error("Inner error");
    } catch (innerError) {
        console.log("Inner catch:", innerError.message);
        throw new Error("Re-thrown as outer error");
    }
} catch (outerError) {
    console.log("Outer catch:", outerError.message);
}
Execution Flow:
  • try - Code that might fail (required)
  • catch - Handles thrown errors (required if no finally)
  • finally - Cleanup code, always runs (optional)
  • Only one of catch or finally is required

Custom Error Classes

Create specialized error types for your application:

// Base application error
class AppError extends Error {
    constructor(message, code, statusCode = 500) {
        super(message);
        this.name = this.constructor.name;
        this.code = code;
        this.statusCode = statusCode;
        Error.captureStackTrace(this, this.constructor);
    }
}

// Validation error
class ValidationError extends AppError {
    constructor(message, field = null) {
        super(message, "VALIDATION_ERROR", 400);
        this.field = field;
    }
}

// Not found error
class NotFoundError extends AppError {
    constructor(resource, id) {
        super(`${resource} with id '${id}' not found`, "NOT_FOUND", 404);
        this.resource = resource;
        this.resourceId = id;
    }
}

// Authentication error
class AuthenticationError extends AppError {
    constructor(message = "Authentication required") {
        super(message, "AUTHENTICATION_ERROR", 401);
    }
}

// Authorization error
class AuthorizationError extends AppError {
    constructor(message = "Access denied") {
        super(message, "AUTHORIZATION_ERROR", 403);
    }

// Usage examples
function findUser(id) {
    const user = database.find(u => u.id === id);

Async Error Handling

Handle errors in asynchronous code:

// Promises with .catch()
fetch("/api/data")
    .then(response => response.json())

    .catch(error => {

// async/await with try...catch
async function fetchUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        
        if (!response.ok) {
            throw new Error(`HTTP error: ${response.status}`);
        }
        
        const user = await response.json();
        return user;
    } catch (error) {
        console.error("Error fetching user:", error);
        throw error;  // Re-throw for caller to handle
    }
}

// Handle multiple async operations
async function fetchAll() {
    const results = await Promise.allSettled([
        fetch("/api/users"),
        fetch("/api/posts"),
        fetch("/api/comments")
    ]);
    
    results.forEach((result, index) => {
        if (result.status === "rejected") {
            console.error(`Request ${index} failed:`, result.reason);
        }
    });
    
    return results
        .filter(r => r.status === "fulfilled")
        .map(r => r.value);
}

// Async error wrapper
function asyncHandler(fn) {
    return (req, res, next) => {
        Promise.resolve(fn(req, res, next)).catch(next);
    };
}

// Express.js style
app.get("/users/:id", asyncHandler(async (req, res) => {
    const user = await findUser(req.params.id);
    res.json(user);
}));

// Catch unhandled promise rejections
process.on("unhandledRejection", (reason, promise) => {
    console.error("Unhandled Rejection at:", promise, "reason:", reason);
});

// In browser
window.addEventListener("unhandledrejection", (event) => {
    console.error("Unhandled rejection:", event.reason);

// Pattern: Error boundaries (React-style)
class ErrorBoundary {
    constructor() {
        this.hasError = false;
        this.error = null;
    }
    
    try(fn) {
        try {
            return fn();
        } catch (error) {
            this.hasError = true;
            this.error = error;
            this.onError(error);
            return null;
        }
    }
    
    onError(error) {
        console.error("Error caught:", error);
        // Render fallback UI
    }
}

// Pattern: Result type (functional approach)
class Result {
    static ok(value) {
        return { success: true, value, error: null };
    }
    
    static fail(error) {
        return { success: false, value: null, error };
    }
}

function divideSafely(a, b) {
    if (b === 0) {
        return Result.fail("Cannot divide by zero");
    }
    return Result.ok(a / b);
}

const result = divideSafely(10, 2);
if (result.success) {
    console.log("Result:", result.value);
} else {
    console.log("Error:", result.error);
}

// Error tracking service integration
class ErrorTracker {
    constructor(apiKey) {
        this.apiKey = apiKey;
        this.setupGlobalHandlers();
    }
    
    setupGlobalHandlers() {
        window.addEventListener("error", (event) => {
            this.capture(event.error || new Error(event.message));
        });
        
        window.addEventListener("unhandledrejection", (event) => {
            this.capture(event.reason);
        });
    }
    
    capture(error, context = {}) {
        const errorData = {
            message: error.message,
            stack: error.stack,
            url: window.location.href,
            userAgent: navigator.userAgent,
            timestamp: new Date().toISOString(),
            ...context
        };
        
        // Send to tracking service
        this.send(errorData);
        
        // Also log locally
        console.error("Captured error:", errorData);
    }
    
    async send(data) {
        try {
            await fetch("https://errors.example.com/api/capture", {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify(data)
            });
        } catch (e) {
            console.error("Failed to send error:", e);
        }
    }
}

// Defensive programming
function processUser(user) {

        throw new ValidationError("Valid email is required", "email");
    }
    
    // Process user
    return normalizeUser(user);
}

// Type checking utility
function assertType(value, type, name = "value") {
    if (typeof value !== type) {
        throw new TypeError(`${name} must be a ${type}, got ${typeof value}`);
    }
}

function assertInstance(value, constructor, name = "value") {
    if (!(value instanceof constructor)) {
        throw new TypeError(`${name} must be an instance of ${constructor.name}`);
    }
}

// Usage
function processData(data) {
    assertType(data, "object", "data");
    assertType(data.items, "object", "data.items");
    assertInstance(data.items, Array, "data.items");
    // Now safe to process
}

// Schema validation
function validateUser(data) {
    const schema = {
        name: { type: "string", required: true, minLength: 2 },
        email: { type: "string", required: true, pattern: /^[^\s@]+@[^\s@]+\.[^\s@]+$/ },
        age: { type: "number", required: false, min: 0, max: 150 }
    };
    
    const errors = [];
    
    for (const [field, rules] of Object.entries(schema)) {
        const value = data[field];
        
        if (rules.required && (value === undefined || value === null || value === "")) {
            errors.push({ field, message: `${field} is required` });
            continue;
        }
        
        if (value !== undefined) {
            if (rules.type && typeof value !== rules.type) {
                errors.push({ field, message: `${field} must be a ${rules.type}` });
            }
            
            if (rules.minLength && value.length < rules.minLength) {
                errors.push({ field, message: `${field} must be at least ${rules.minLength} characters` });
            }
            
            if (rules.pattern && !rules.pattern.test(value)) {
                errors.push({ field, message: `${field} has invalid format` });
            }
            
            if (rules.min !== undefined && value < rules.min) {
                errors.push({ field, message: `${field} must be at least ${rules.min}` });
            }
            
            if (rules.max !== undefined && value > rules.max) {
                errors.push({ field, message: `${field} must be at most ${rules.max}` });
            }
        }
    }
    
    return errors.length > 0 ? { valid: false, errors } : { valid: true };
}

// Usage
const result = validateUser({
    name: "J",
    email: "invalid-email",
    age: 200
});
console.log(result);
// { valid: false, errors: [...] }

Logging and Debugging

Track errors effectively:

// Logger utility
class Logger {
    constructor(level = "info") {
        this.level = level;
        this.levels = { debug: 0, info: 1, warn: 2, error: 3 };
    }
    
    shouldLog(level) {
        return this.levels[level] >= this.levels[this.level];
    }
    
    formatMessage(level, message, data = {}) {
        return {
            timestamp: new Date().toISOString(),
            level,
            message,
            ...data
        };
    }
    
    debug(message, data) {
        if (this.shouldLog("debug")) {
            console.debug(this.formatMessage("debug", message, data));
        }
    }
    
    info(message, data) {
        if (this.shouldLog("info")) {
            console.info(this.formatMessage("info", message, data));
        }
    }
    
    warn(message, data) {
        if (this.shouldLog("warn")) {
            console.warn(this.formatMessage("warn", message, data));
        }
    }
    
    error(message, error = null, data = {}) {
        if (this.shouldLog("error")) {
            const formatted = this.formatMessage("error", message, {
                ...data,
                error: error ? { message: error.message, stack: error.stack } : null
            });
            console.error(formatted);
        }
    }
}

Error Recovery Strategies

Handle errors gracefully and recover:

// Retry with backoff
async function retryWithBackoff(fn, maxRetries = 3, baseDelay = 1000) {
    let lastError;
    
    for (let attempt = 0; attempt < maxRetries; attempt++) {
        try {
            return await fn();
        } catch (error) {
            lastError = error;
            
            if (attempt < maxRetries - 1) {
                const delay = baseDelay * Math.pow(2, attempt);
                console.log(`Attempt ${attempt + 1} failed, retrying in ${delay}ms`);
                await new Promise(r => setTimeout(r, delay));
            }
        }
    }
    
    throw lastError;
}

// Circuit breaker pattern
class CircuitBreaker {
    constructor(fn, options = {}) {
        this.fn = fn;
        this.failureThreshold = options.failureThreshold || 5;
        this.resetTimeout = options.resetTimeout || 60000;
        this.failures = 0;
        this.state = "closed";  // closed, open, half-open
        this.lastFailure = null;
    }
    
    async execute(...args) {
        if (this.state === "open") {
            if (Date.now() - this.lastFailure > this.resetTimeout) {
                this.state = "half-open";
            } else {
                throw new Error("Circuit breaker is open");
            }
        }
        
        try {
            const result = await this.fn(...args);
            this.onSuccess();
            return result;
        } catch (error) {
            this.onFailure();
            throw error;
        }
    }
    
    onSuccess() {
        this.failures = 0;
        this.state = "closed";
    }
    
    onFailure() {
        this.failures++;
        this.lastFailure = Date.now();
        
        if (this.failures >= this.failureThreshold) {
            this.state = "open";
        }
    }
}

// Exercise: Build a robust form validation system
class FormValidator {
    constructor(schema) {
        this.schema = schema;
        this.errors = {};
    }
    
    validate(data) {
        this.errors = {};
        
        for (const [field, rules] of Object.entries(this.schema)) {
            const value = data[field];
            
            // Required check
            if (rules.required && this.isEmpty(value)) {
                this.addError(field, `${this.formatFieldName(field)} is required`);
                continue;
            }
            
            // Skip other validations if empty and not required
            if (this.isEmpty(value) && !rules.required) continue;
            
            // Type validation
            if (rules.type && !this.checkType(value, rules.type)) {
                this.addError(field, `${this.formatFieldName(field)} must be a ${rules.type}`);
            }
            
            // Custom validators
            if (rules.validators) {
                for (const validator of rules.validators) {
                    const result = validator(value, data);
                    if (result !== true) {
                        this.addError(field, result);
                        break;
                    }
                }
            }
        }
        
        return Object.keys(this.errors).length === 0;
    }
    
    isEmpty(value) {
        return value === undefined || value === null || value === "";
    }
    
    checkType(value, type) {
        switch (type) {
            case "string": return typeof value === "string";
            case "number": return typeof value === "number" && !isNaN(value);
            case "boolean": return typeof value === "boolean";
            case "array": return Array.isArray(value);
            case "email": return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
            default: return true;
        }
    }
    
    addError(field, message) {
        if (!this.errors[field]) {
            this.errors[field] = [];
        }
        this.errors[field].push(message);
    }
    
    formatFieldName(field) {
        return field.charAt(0).toUpperCase() + field.slice(1).replace(/([A-Z])/g, ' $1');
    }
    
    getErrors() {
        return this.errors;
    }
    
    hasErrors() {
        return Object.keys(this.errors).length > 0;
    }
}
