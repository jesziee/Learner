Lesson 5: Loops

Learn how to repeat code efficiently using different types of loops.

What are Loops?

Loops allow you to execute the same code multiple times without writing it over and over. Instead of writing 100 lines of code to print something 100 times, you can use a loop to do it in just a few lines. Loops are one of the most powerful tools in programming!

Why Loops Matter: Imagine processing a list of 1000 users, or checking every character in a document. Without loops, you'd need to write thousands of lines of repetitive code. Loops solve this elegantly.

The for Loop

The most commonly used loop - perfect when you know how many times to repeat:

// Basic for loop structure
for (let i = 0; i < 5; i++) {
    console.log("Iteration number: " + i);
}

// Output:
// Iteration number: 0
// Iteration number: 1
// Iteration number: 2
// Iteration number: 3
// Iteration number: 4
For Loop Anatomy:
  • let i = 0 - Initialization: Starting value (runs once at the beginning)
  • i < 5 - Condition: Loop continues while this is true
  • i++ - Update: Runs after each iteration (increments i)
  • { } - Loop body: Code to repeat

How the Loop Executes

for (let i = 1; i <= 3; i++) {
    console.log("i is now: " + i);
}

console.log("Loop finished!");

// Step by step:
// 1. i = 1, condition (1 <= 3) is true, print "i is now: 1", i becomes 2
// 2. i = 2, condition (2 <= 3) is true, print "i is now: 2", i becomes 3
// 3. i = 3, condition (3 <= 3) is true, print "i is now: 3", i becomes 4
// 4. i = 4, condition (4 <= 3) is false, loop ends
// 5. Print "Loop finished!"

Counting Down

// Countdown from 5 to 1
for (let i = 5; i >= 1; i--) {
    console.log(i);
}
console.log("Blast off!");

// Output: 5, 4, 3, 2, 1, Blast off!

// Counting by 2s
for (let i = 0; i <= 10; i += 2) {
    console.log(i);
}
// Output: 0, 2, 4, 6, 8, 10

The while Loop

Use when you don't know exactly how many times to loop:

// While loop structure
let count = 0;

while (count < 5) {
    console.log("Count is: " + count);
    count++;
}

// Output: Count is: 0, 1, 2, 3, 4
While Loop Anatomy:
  • while (condition) - Loop continues while condition is true
  • Condition is checked BEFORE each iteration
  • You must update the condition variable inside the loop
  • If you forget to update, you get an infinite loop!

Practical Example: User Input

// Keep asking until valid input
let userInput = "";
let attempts = 0;

// Simulating user input (in real code, you'd use prompt())
let inputs = ["", "x", "abc", "password"];

while (userInput !== "password" && attempts < inputs.length) {
    userInput = inputs[attempts];
    attempts++;
    console.log("Attempt " + attempts + ": " + userInput);
}

if (userInput === "password") {
    console.log("Access granted!");
}

// Output shows the progression until correct password

Watch Out: Infinite Loops

// ⚠️ DANGER: Infinite loop!
// let x = 0;
// while (x < 5) {
//     console.log(x);
//     // Forgot to increment x! Browser will freeze!
// }

// Correct version:
let x = 0;
while (x < 5) {
    console.log(x);
    x++;  // Don't forget this!
}

The do...while Loop

Similar to while, but always runs at least once:

// do...while structure
let number = 10;

do {
    console.log("Number is: " + number);
    number++;
} while (number < 5);

// Output: "Number is: 10"
// Even though 10 < 5 is false, it runs once!

// Compare with while loop:
let number2 = 10;
while (number2 < 5) {
    console.log("This never prints");
}
// Nothing printed - condition was false from the start
Key Difference:
  • while - Checks condition FIRST, might never run
  • do...while - Runs code FIRST, then checks condition
  • Use do...when you need at least one execution

Breaking Out of Loops

Use break to exit a loop early:

// Find the first number divisible by 7
for (let i = 1; i <= 100; i++) {
    if (i % 7 === 0) {
        console.log("Found it: " + i);
        break;  // Exit the loop immediately
    }
}
// Output: "Found it: 7"

// Search through an array
let names = ["John", "Jane", "Bob", "Alice", "Charlie"];
let searchName = "Bob";
let found = false;

for (let i = 0; i < names.length; i++) {
    if (names[i] === searchName) {
        console.log("Found " + searchName + " at index " + i);
        found = true;
        break;
    }
}

if (!found) {
    console.log(searchName + " not found");
}

Skipping Iterations

Use continue to skip to the next iteration:

// Print only odd numbers
for (let i = 1; i <= 10; i++) {
    if (i % 2 === 0) {
        continue;  // Skip even numbers
    }
    console.log(i);
}
// Output: 1, 3, 5, 7, 9

// Skip invalid data
let scores = [85, -1, 92, 78, -1, 95];

for (let i = 0; i < scores.length; i++) {
    if (scores[i] === -1) {
        continue;  // Skip missing scores
    }
    console.log("Score: " + scores[i]);
}
// Output: Score: 85, Score: 92, Score: 78, Score: 95
Break vs Continue:
  • break - Exits the entire loop
  • continue - Skips to the next iteration
  • Both make loops more flexible and efficient

Nested Loops

Loops inside loops - useful for multi-dimensional data:

// Multiplication table
for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 3; j++) {
        console.log(i + " x " + j + " = " + (i * j));
    }
    console.log("---");  // Separator between rows
}

// Output:
// 1 x 1 = 1, 1 x 2 = 2, 1 x 3 = 3
// ---
// 2 x 1 = 2, 2 x 2 = 4, 2 x 3 = 6
// ---
// 3 x 1 = 3, 3 x 2 = 6, 3 x 3 = 9
// ---
How Nested Loops Work:
  • Outer loop runs 3 times (i = 1, 2, 3)
  • For each outer loop, inner loop runs 3 times (j = 1, 2, 3)
  • Total: 3 x 3 = 9 iterations
  • Inner loop completes before outer loop continues

Working with 2D Arrays

// A 2D array (array of arrays)
let matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
];

// Print all values
for (let row = 0; row < matrix.length; row++) {
    for (let col = 0; col < matrix[row].length; col++) {
        console.log("matrix[" + row + "][" + col + "] = " + matrix[row][col]);
    }
}

// Find the sum of all numbers
let sum = 0;
for (let row = 0; row < matrix.length; row++) {
    for (let col = 0; col < matrix[row].length; col++) {
        sum += matrix[row][col];
    }
}
console.log("Sum: " + sum);  // 45

Looping Through Arrays

Arrays and loops are a perfect combination:

let fruits = ["Apple", "Banana", "Orange", "Grape"];

// Method 1: Classic for loop
for (let i = 0; i < fruits.length; i++) {
    console.log(fruits[i]);
}

// Method 2: for...of loop (cleaner!)
for (let fruit of fruits) {
    console.log(fruit);
}

// Method 3: forEach method
fruits.forEach(function(fruit) {
    console.log(fruit);
});

// All three output: Apple, Banana, Orange, Grape
Which to Use:
  • for - When you need the index number
  • for...of - When you just need the values (most common)
  • forEach - When using functional programming style

for...of vs for...in

let colors = ["Red", "Green", "Blue"];

// for...of gives VALUES
for (let color of colors) {
    console.log(color);  // "Red", "Green", "Blue"
}

// for...in gives INDICES (usually not what you want for arrays)
for (let index in colors) {
    console.log(index);  // "0", "1", "2"
    console.log(colors[index]);  // "Red", "Green", "Blue"
}
Best Practice: Use for...of for arrays when you need the values. Use for...in for objects when you need property names.

Looping Through Objects

Use for...in to iterate over object properties:

let person = {
    name: "John",
    age: 30,
    city: "New York",
    job: "Developer"
};

// Loop through all properties
for (let key in person) {
    console.log(key + ": " + person[key]);
}

// Output:
// name: John
// age: 30
// city: New York
// job: Developer

// Get all keys
let keys = Object.keys(person);
console.log(keys);  // ["name", "age", "city", "job"]

// Get all values
let values = Object.values(person);
console.log(values);  // ["John", 30, "New York", "Developer"]

Practical Examples

Sum of Numbers

// Calculate sum of 1 to 100
let total = 0;
for (let i = 1; i <= 100; i++) {
    total += i;
}
console.log("Sum: " + total);  // 5050

// Shortcut using formula
let n = 100;
let formulaSum = (n * (n + 1)) / 2;
console.log("Formula result: " + formulaSum);  // 5050

Finding Maximum

let numbers = [23, 45, 12, 67, 34, 89, 56];
let max = numbers[0];

for (let num of numbers) {
    if (num > max) {
        max = num;
    }
}

console.log("Maximum: " + max);  // 89

Filtering Data

let ages = [16, 21, 14, 35, 18, 12, 42];
let adults = [];

for (let age of ages) {
    if (age >= 18) {
        adults.push(age);
    }
}

console.log("Adults:", adults);  // [21, 35, 18, 42]

Building HTML

let products = [
    { name: "Laptop", price: 999 },
    { name: "Phone", price: 699 },
    { name: "Tablet", price: 499 }
];

let html = "
    ";

for (let product of products) {
    html += `
  • ${product.name}: $${product.price}
    • `;
}

html += "
";
console.log(html);
// 
  • Laptop: $999
  • Phone: $699
  • Tablet: $499

// Exercise 3: Reverse a string
let original = "JavaScript";
let reversed = "";
for (let i = original.length - 1; i >= 0; i--) {
    reversed += original[i];
}
console.log(original + " reversed is " + reversed);

        console.log("Fizz");

        console.log("Buzz");
    } else {

    }
}
