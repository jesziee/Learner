Why Best Practices Matter

Writing clean, maintainable JavaScript is crucial for professional development. Good practices make your code easier to read, debug, and scale. They help teams collaborate effectively and reduce technical debt over time.

Code Organization

Organizing your code properly is the foundation of maintainable JavaScript. A well-structured codebase is easier to navigate, debug, and extend.

Module Pattern Example:

//userManager.js - A module for user operations
const UserManager = (function() {
    // Private variables and functions
    let users = [];
    
    function validateUser(user) {
        return user.name && user.email;
    }
    
    function generateId() {
        return Date.now().toString(36) + Math.random().toString(36).substr(2);
    }
    
    // Public API
    return {
        addUser: function(userData) {
            if (!validateUser(userData)) {
                throw new Error('Invalid user data');
            }
            const user = {
                id: generateId(),
                ...userData,
                createdAt: new Date()
            };
            users.push(user);
            return user;
        },
        
        getUser: function(id) {
            return users.find(user => user.id === id);
        },
        
        getAllUsers: function() {
            return [...users]; // Return a copy, not the original
        },
        
        deleteUser: function(id) {
            const index = users.findIndex(user => user.id === id);
            if (index > -1) {
                users.splice(index, 1);
                return true;
            }
            return false;
        }
    };
})();

// Usage
const newUser = UserManager.addUser({
    name: 'John Doe',
    email: 'john@example.com'
});
console.log(newUser);

Line 1: We create an Immediately Invoked Function Expression (IIFE) that returns an object with public methods.

Line 3: The users array is private - it cannot be accessed directly from outside the module.

Lines 5-7: Private helper function to validate user data before adding.

Lines 9-11: Private function to generate unique IDs using timestamp and random string.

Lines 14-35: The returned object contains only the public methods, creating a clean API.

Line 33: We return a spread copy of the users array to prevent external modification of the internal state.

Error Handling Best Practices

Proper error handling makes your applications more robust and easier to debug. Always anticipate potential failures and handle them gracefully.

Robust Error Handling:

class AppError extends Error {
    constructor(message, statusCode, code) {
        super(message);
        this.statusCode = statusCode;
        this.code = code;
        this.timestamp = new Date().toISOString();
    }
}

class ValidationError extends AppError {
    constructor(message, field) {
        super(message, 400, 'VALIDATION_ERROR');
        this.field = field;
    }
}

class NotFoundError extends AppError {
    constructor(resource) {
        super(`${resource} not found`, 404, 'NOT_FOUND');
        this.resource = resource;
    }
}

// Usage in async function
async function fetchUserData(userId) {
    if (!userId) {
        throw new ValidationError('User ID is required', 'userId');
    }
    
    try {
        const response = await fetch(`/api/users/${userId}`);
        
        if (response.status === 404) {
            throw new NotFoundError('User');
        }
        
        if (!response.ok) {
            throw new AppError(
                'Failed to fetch user',
                response.status,
                'FETCH_ERROR'
            );
        }
        
        return await response.json();
    } catch (error) {
        // Log for debugging (in production, use a logging service)
        console.error({
            message: error.message,
            code: error.code,
            timestamp: error.timestamp,
            stack: error.stack
        });
        
        // Re-throw custom errors, wrap unexpected errors
        if (error instanceof AppError) {
            throw error;
        }
        
        throw new AppError(
            'An unexpected error occurred',
            500,
            'INTERNAL_ERROR'
        );
    }
}

Lines 1-7: Custom error class that extends the built-in Error with additional properties like status code and error code.

Lines 9-14: Specific error type for validation failures with the field that failed validation.

Lines 16-21: Specific error type for resource not found scenarios.

Lines 24-26: Input validation at the start of the function - fail fast with clear error messages.

Lines 31-33: Convert HTTP 404 to our custom NotFoundError for consistent error handling.

Lines 50-52: Re-throw custom errors as-is, preserving their specific information.

Performance Optimization

Writing performant JavaScript requires understanding how the engine executes your code. Here are key techniques for optimization.

Performance Patterns:

// 1. Debouncing - Limit how often a function runs
function debounce(func, wait) {
    let timeout;
    return function executedFunction(...args) {
        const later = () => {
            clearTimeout(timeout);
            func(...args);
        };
        clearTimeout(timeout);
        timeout = setTimeout(later, wait);
    };
}

// Usage: Search input that waits for user to stop typing
const searchInput = document.getElementById('search');
const debouncedSearch = debounce(async (query) => {
    const results = await fetchSearchResults(query);
    displayResults(results);
}, 300);

searchInput.addEventListener('input', (e) => {
    debouncedSearch(e.target.value);
});

// 2. Throttling - Limit function to run at most once per period
function throttle(func, limit) {
    let inThrottle;
    return function(...args) {
        if (!inThrottle) {
            func.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

// Usage: Scroll handler that runs at most every 100ms
window.addEventListener('scroll', throttle(() => {
    updateScrollPosition();
}, 100));

// 3. Memoization - Cache expensive function results
function memoize(fn) {
    const cache = new Map();
    return function(...args) {
        const key = JSON.stringify(args);
        if (cache.has(key)) {
            return cache.get(key);
        }
        const result = fn.apply(this, args);
        cache.set(key, result);
        return result;
    };
}

// Usage: Cache expensive calculations
const expensiveCalculation = memoize((n) => {
    // Simulate expensive operation
    let result = 0;
    for (let i = 0; i < n; i++) {
        result += Math.sqrt(i) * Math.sin(i);
    }
    return result;
});

// 4. Lazy Loading - Load content when needed
const lazyLoadImages = () => {
    const images = document.querySelectorAll('img[data-src]');
    
    const imageObserver = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                const img = entry.target;
                img.src = img.dataset.src;
                img.removeAttribute('data-src');
                imageObserver.unobserve(img);
            }
        });
    });
    
    images.forEach(img => imageObserver.observe(img));
};

Lines 2-11: Debounce function that delays execution until after a pause in calls. Great for search inputs.

Lines 13-21: Practical usage of debounce for a search input that waits 300ms after the user stops typing.

Lines 24-33: Throttle function that ensures a function runs at most once per specified time period.

Lines 35-38: Scroll event throttling to prevent excessive function calls during scrolling.

Lines 41-52: Memoization implementation that caches function results based on arguments.

Lines 66-79: Lazy loading using IntersectionObserver to load images only when they're about to be visible.

Security Best Practices

Security should be built into your code from the start. Here are essential patterns for writing secure JavaScript applications.

Security Patterns:

// 1. Input Sanitization
function sanitizeInput(input) {
    if (typeof input !== 'string') return '';
    
    return input
        .replace(/&/g, '&')
        .replace(//g, '>')
        .replace(/"/g, '"')
        .replace(/'/g, ''');
}

// 2. Content Security Policy Helper
const CSP = {
    directives: {
        'default-src': ["'self'"],
        'script-src': ["'self'", 'https://trusted.cdn.com'],
        'style-src': ["'self'", 'https://fonts.googleapis.com'],
        'img-src': ["'self'", 'data:', 'https:'],
        'connect-src': ["'self'", 'https://api.example.com']
    },
    
    generateHeader() {
        return Object.entries(this.directives)
            .map(([key, values]) => `${key} ${values.join(' ')}`)
            .join('; ');
    }
};

// 3. Secure Token Storage
const SecureStorage = {
    set(key, value, options = {}) {
        const item = {
            value,
            expiry: options.expiry ? Date.now() + options.expiry : null
        };
        // For sensitive data, consider using sessionStorage instead
        const storage = options.session ? sessionStorage : localStorage;
        storage.setItem(key, JSON.stringify(item));
    },
    
    get(key, options = {}) {
        const storage = options.session ? sessionStorage : localStorage;
        const itemStr = storage.getItem(key);
        
        if (!itemStr) return null;
        
        const item = JSON.parse(itemStr);
        
        if (item.expiry && Date.now() > item.expiry) {
            storage.removeItem(key);
            return null;
        }
        
        return item.value;
    },
    
    remove(key, options = {}) {
        const storage = options.session ? sessionStorage : localStorage;
        storage.removeItem(key);
    }
};

// 4. XSS Prevention in DOM Manipulation
function safeSetHTML(element, content) {
    // Use textContent for plain text (automatically escaped)
    element.textContent = content;
}

function safeCreateElement(tag, attributes = {}, children = []) {
    const element = document.createElement(tag);
    
    // Safely set attributes
    Object.entries(attributes).forEach(([key, value]) => {
        if (key.startsWith('on')) {
            // Prevent inline event handlers
            console.warn(`Inline event handler '${key}' not allowed`);
            return;
        }
        if (key === 'href' || key === 'src') {
            // Validate URLs
            try {
                const url = new URL(value, window.location.origin);
                if (url.protocol !== 'http:' && 
                    url.protocol !== 'https:' && 
                    url.protocol !== 'mailto:') {
                    console.warn(`Invalid protocol for ${key}`);
                    return;
                }
            } catch (e) {
                console.warn(`Invalid URL: ${value}`);
                return;
            }
        }
        element.setAttribute(key, value);
    });
    
    // Safely add children
    children.forEach(child => {
        if (typeof child === 'string') {
            element.appendChild(document.createTextNode(child));
        } else if (child instanceof Node) {
            element.appendChild(child);
        }
    });
    
    return element;
}

Lines 13-30: Content Security Policy configuration object that helps define what resources can be loaded.

Lines 33-61: Secure storage wrapper with expiry support and session storage option for sensitive data.

Lines 64-66: Using textContent instead of innerHTML automatically escapes HTML.

Lines 69-98: Safe element creation that validates URLs and prevents dangerous inline event handlers.

Lines 91-93: Using createTextNode ensures text content is properly escaped.

Final Project: Task Manager Application

Let's build a complete task manager application that demonstrates all the concepts we've learned. This project combines DOM manipulation, local storage, error handling, and best practices.

Complete Task Manager:

// taskManager.js - Complete Task Manager Application

// ==================== Models ====================
class Task {
    constructor(title, description = '', priority = 'medium') {
        this.id = crypto.randomUUID();
        this.title = title;
        this.description = description;
        this.priority = priority;
        this.completed = false;
        this.createdAt = new Date().toISOString();
        this.completedAt = null;
    }
    
    complete() {
        this.completed = true;
        this.completedAt = new Date().toISOString();
    }
    
    uncomplete() {
        this.completed = false;
        this.completedAt = null;
    }
}

// ==================== Storage Service ====================
class StorageService {
    constructor(key) {
        this.key = key;
    }
    
    get() {
        try {
            const data = localStorage.getItem(this.key);
            return data ? JSON.parse(data) : [];
        } catch (error) {
            console.error('Storage read error:', error);
            return [];
        }
    }
    
    save(data) {
        try {
            localStorage.setItem(this.key, JSON.stringify(data));
            return true;
        } catch (error) {
            console.error('Storage write error:', error);
            return false;
        }
    }
}

// ==================== Task Manager ====================
class TaskManager {
    constructor(storageKey = 'tasks') {
        this.storage = new StorageService(storageKey);
        this.tasks = this.loadTasks();
        this.listeners = new Set();
    }
    
    loadTasks() {
        const data = this.storage.get();
        return data.map(t => Object.assign(new Task(''), t));
    }
    
    saveTasks() {
        this.storage.save(this.tasks);
        this.notifyListeners();
    }
    
    subscribe(callback) {
        this.listeners.add(callback);
        return () => this.listeners.delete(callback);
    }
    
    notifyListeners() {
        this.listeners.forEach(callback => callback(this.tasks));
    }
    
    addTask(title, description, priority) {
        const task = new Task(title, description, priority);
        this.tasks.unshift(task);
        this.saveTasks();
        return task;
    }
    
    deleteTask(id) {
        const index = this.tasks.findIndex(t => t.id === id);
        if (index > -1) {
            this.tasks.splice(index, 1);
            this.saveTasks();
            return true;
        }
        return false;
    }
    
    toggleTask(id) {
        const task = this.tasks.find(t => t.id === id);
        if (task) {
            task.completed ? task.uncomplete() : task.complete();
            this.saveTasks();
            return task;
        }
        return null;
    }
    
    getStats() {
        const total = this.tasks.length;
        const completed = this.tasks.filter(t => t.completed).length;
        const pending = total - completed;
        return { total, completed, pending };
    }
}

// ==================== UI Controller ====================
class TaskUI {
    constructor(taskManager) {
        this.manager = taskManager;
        this.elements = {};
        this.init();
    }
    
    init() {
        this.cacheElements();
        this.bindEvents();
        this.manager.subscribe(() => this.render());
        this.render();
    }
    
    cacheElements() {
        this.elements = {
            form: document.getElementById('task-form'),
            titleInput: document.getElementById('task-title'),
            descInput: document.getElementById('task-description'),
            prioritySelect: document.getElementById('task-priority'),
            taskList: document.getElementById('task-list'),
            stats: document.getElementById('task-stats'),
            clearCompleted: document.getElementById('clear-completed')
        };
    }
    
    bindEvents() {
        this.elements.form.addEventListener('submit', (e) => {
            e.preventDefault();
            this.handleAddTask();
        });
        
        this.elements.taskList.addEventListener('click', (e) => {
            this.handleTaskAction(e);
        });
        
        this.elements.clearCompleted.addEventListener('click', () => {
            this.handleClearCompleted();
        });
    }
    
    handleAddTask() {
        const title = this.elements.titleInput.value.trim();
        const description = this.elements.descInput.value.trim();
        const priority = this.elements.prioritySelect.value;
        
        if (!title) {
            this.showError('Please enter a task title');
            return;
        }
        
        this.manager.addTask(title, description, priority);
        this.elements.form.reset();
        this.elements.titleInput.focus();
    }
    
    handleTaskAction(e) {
        const taskEl = e.target.closest('[data-task-id]');
        if (!taskEl) return;
        
        const taskId = taskEl.dataset.taskId;
        
        if (e.target.matches('.toggle-btn')) {
            this.manager.toggleTask(taskId);
        } else if (e.target.matches('.delete-btn')) {
            this.manager.deleteTask(taskId);
        }
    }
    
    handleClearCompleted() {
        const completed = this.manager.tasks.filter(t => t.completed);
        completed.forEach(t => this.manager.deleteTask(t.id));
    }
    
    render() {
        this.renderTasks();
        this.renderStats();
    }
    
    renderTasks() {
        const html = this.manager.tasks.map(task => this.createTaskHTML(task)).join('');
        this.elements.taskList.innerHTML = html || '
No tasks yet. Add one above!
';
    }
    
    createTaskHTML(task) {
        const priorityClass = `priority-${task.priority}`;
        const completedClass = task.completed ? 'completed' : '';
        
        return `
            

                

                    
${this.escapeHTML(task.title)}

                    ${task.description ? `
${this.escapeHTML(task.description)}
` : ''}
                    

                        ${task.priority}
                        ${new Date(task.createdAt).toLocaleDateString()}
                    

                

                

                    
                    
                

            

        `;
    }
    
    renderStats() {
        const stats = this.manager.getStats();
        this.elements.stats.innerHTML = `
            Total: ${stats.total}
            Completed: ${stats.completed}
            Pending: ${stats.pending}
        `;
    }
    
    escapeHTML(str) {
        const div = document.createElement('div');
        div.textContent = str;
        return div.innerHTML;
    }
    
    showError(message) {
        // Simple error display - in production, use a proper notification system
        alert(message);
    }
}

// ==================== Initialize Application ====================
document.addEventListener('DOMContentLoaded', () => {
    const manager = new TaskManager();
    const ui = new TaskUI(manager);
});
