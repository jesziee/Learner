Lesson 9: DOM Manipulation

Learn to interact with web pages dynamically using the Document Object Model.

What is the DOM?

The Document Object Model (DOM) is a programming interface for web documents. It represents the page as a tree of objects that JavaScript can manipulate. When a web page loads, the browser creates a DOM of the page, which allows JavaScript to access and change the HTML elements, attributes, and CSS styles dynamically.

Key Concept: Think of the DOM as a bridge between your HTML and JavaScript. HTML provides the structure, and through the DOM, JavaScript can read, modify, add, or delete elements and content.

Selecting Elements

Before manipulating elements, you need to select them:

Modern Methods (Preferred)

// querySelector - first matching element
const header = document.querySelector("h1");
const container = document.querySelector("#container");
const text = document.querySelector(".text");
const firstParagraph = document.querySelector("p.intro");

// querySelectorAll - all matching elements (NodeList)
const allParagraphs = document.querySelectorAll("p");
const allItems = document.querySelectorAll(".list-item");

// Iterate over NodeList
allParagraphs.forEach(paragraph => {
    console.log(paragraph.textContent);
});

// Convert NodeList to Array for more methods
const paragraphsArray = Array.from(allParagraphs);
const filtered = paragraphsArray.filter(p => p.textContent.length > 50);

Traditional Methods

// getElementById - fastest for IDs
const element = document.getElementById("myId");

// getElementsByClassName - returns HTMLCollection (live)
const items = document.getElementsByClassName("item");

// getElementsByTagName - returns HTMLCollection (live)
const divs = document.getElementsByTagName("div");

// Convert HTMLCollection to Array
const itemsArray = Array.from(items);

// Live collection updates automatically
const liveList = document.getElementsByClassName("item");
console.log(liveList.length);  // 3
// Add new element with class "item"
console.log(liveList.length);  // 4 (automatically updated)
querySelector vs getElementById:
  • querySelector - More flexible, uses CSS selectors, returns first match
  • getElementById - Faster, only for IDs, single element
  • querySelectorAll - Returns static NodeList
  • getElementsBy* - Returns live HTMLCollection

Traversing the DOM

Navigate between related elements:

const element = document.querySelector(".container");

// Parent element
const parent = element.parentElement;
const closestParent = element.closest(".wrapper");  // Find ancestor

// Children
const children = element.children;        // HTMLCollection of child elements
const firstChild = element.firstElementChild;
const lastChild = element.lastElementChild;

// Siblings
const nextSibling = element.nextElementSibling;
const previousSibling = element.previousElementSibling;

// Check if element has children
const hasChildren = element.hasChildNodes();

// Example: Get all siblings
function getAllSiblings(element) {
    const siblings = [];
    let sibling = element.parentElement.firstElementChild;
    
    while (sibling) {
        if (sibling !== element) {
            siblings.push(sibling);
        }
        sibling = sibling.nextElementSibling;
    }
    
    return siblings;
}

Modifying Element Content

Change what's inside elements:

const element = document.querySelector("#content");

// textContent - gets/sets text only
console.log(element.textContent);   // Get text
element.textContent = "New text";   // Set text (safe from XSS)

// innerHTML - gets/sets HTML
console.log(element.innerHTML);     // Get HTML including tags
element.innerHTML = "Bold text";  // Parses HTML

// outerHTML - includes the element itself
console.log(element.outerHTML);     // Includes the element tag
element.outerHTML = "
Replaced
";  // Replaces element

// innerText - similar to textContent but respects CSS styling
console.log(element.innerText);     // Respects visibility: hidden

// Differences in action
const div = document.querySelector("#example");
div.innerHTML = "Hidden";

console.log(div.textContent);  // "Hidden" (includes hidden text)
console.log(div.innerText);    // "" (respects display:none)
Security Warning: Never use innerHTML with user input! It can lead to XSS (Cross-Site Scripting) attacks. Use textContent for user-provided content.

Modifying Attributes

Work with element attributes:

const link = document.querySelector("a");

// Get attribute
const href = link.getAttribute("href");
const target = link.getAttribute("target");

// Set attribute
link.setAttribute("href", "https://example.com");
link.setAttribute("target", "_blank");

// Remove attribute
link.removeAttribute("target");

// Check if attribute exists
const hasTarget = link.hasAttribute("target");  // true/false

// Get all attributes
const attrs = link.attributes;
for (let attr of attrs) {
    console.log(attr.name, attr.value);
}

// Special attributes have direct properties
const input = document.querySelector("input");
input.value = "New value";        // Value property
input.disabled = true;            // Boolean attribute
input.checked = true;             // For checkboxes
input.id = "newId";               // ID property

// data-* attributes
const card = document.querySelector(".card");
card.dataset.userId = "123";       // Sets data-user-id="123"
card.dataset.userName = "Alice";   // Sets data-user-name="Alice"
console.log(card.dataset.userId);  // "123"
console.log(card.dataset.userName); // "Alice"

Modifying Classes

Manipulate CSS classes:

const element = document.querySelector(".box");

// classList methods (preferred)
element.classList.add("active");           // Add class
element.classList.remove("hidden");        // Remove class
element.classList.toggle("expanded");      // Toggle class
element.classList.replace("old", "new");   // Replace class

// Multiple classes
element.classList.add("class1", "class2", "class3");
element.classList.remove("class1", "class2");

// Check if has class
const isActive = element.classList.contains("active");  // true/false

// className property (gets/sets all classes)
element.className = "box active expanded";
console.log(element.className);  // "box active expanded"

// Example: Toggle menu
const menu = document.querySelector(".menu");
const button = document.querySelector(".menu-toggle");

button.addEventListener("click", () => {
    menu.classList.toggle("open");
});

// Example: Add class based on condition
const score = 85;
const resultElement = document.querySelector(".result");

if (score >= 70) {
    resultElement.classList.add("pass");

    resultElement.classList.add("fail");
}

Modifying Styles

Change CSS styles directly:

const element = document.querySelector(".box");

// Inline styles via style property
element.style.color = "red";
element.style.backgroundColor = "blue";  // camelCase, not kebab-case
element.style.fontSize = "20px";
element.style.padding = "10px 20px";

// Remove inline style
element.style.color = "";  // Empty string removes it

// Multiple styles
Object.assign(element.style, {
    color: "white",
    backgroundColor: "navy",
    padding: "20px",
    borderRadius: "5px"
});

// Get computed style (actual applied styles)
const computed = window.getComputedStyle(element);
console.log(computed.color);           // Actual color
console.log(computed.backgroundColor); // Actual background

// CSS custom properties (variables)
element.style.setProperty("--main-color", "purple");
const mainColor = element.style.getPropertyValue("--main-color");

// cssText - set all styles at once
element.style.cssText = "color: white; background: navy; padding: 10px;";

// Toggle visibility
element.style.display = "none";        // Hide
element.style.display = "block";       // Show
element.style.visibility = "hidden";   // Hide (keeps space)
element.style.visibility = "visible";  // Show

// Better approach: Use classes for styles
// CSS: .hidden { display: none; }
element.classList.add("hidden");
element.classList.remove("hidden");
Best Practice: Prefer CSS classes over inline styles for maintainability. Use inline styles only for dynamic values that can't be in CSS.

Inserting Elements

Add elements to the DOM:

const parent = document.querySelector(".container");
const newElement = document.createElement("div");
newElement.textContent = "New Element";

// appendChild - add as last child
parent.appendChild(newElement);

// append - add multiple elements at end (newer)
parent.append(newElement, "Some text", anotherElement);

// prepend - add at beginning
parent.prepend(newElement);

// insertBefore - insert before specific element
const reference = document.querySelector(".reference");
parent.insertBefore(newElement, reference);

// Modern insertion methods
const box = document.querySelector(".box");

// Insert relative to element
box.before(newElement);           // Before the element
box.after(newElement);            // After the element
box.prepend(newElement);          // Inside, at beginning
box.append(newElement);           // Inside, at end

// replaceWith - replace element
box.replaceWith(newElement);

// insertAdjacentHTML - insert HTML string
const container = document.querySelector(".container");
container.insertAdjacentHTML("beforebegin", "
Before container
");
container.insertAdjacentHTML("afterbegin", "
First child
");
container.insertAdjacentHTML("beforeend", "
Last child
");
container.insertAdjacentHTML("afterend", "
After container
");

// insertAdjacentElement - insert element
container.insertAdjacentElement("beforeend", newElement);
Insertion Positions:
  • beforebegin - Before the element itself
  • afterbegin - Inside, before first child
  • beforeend - Inside, after last child
  • afterend - After the element itself

Removing Elements

Delete elements from the DOM:

// Modern method - remove element itself
const element = document.querySelector(".element-to-remove");
element.remove();

// Traditional method - remove from parent
const parent = document.querySelector(".parent");
const child = document.querySelector(".child");
parent.removeChild(child);

// Remove all children
const container = document.querySelector(".container");
container.innerHTML = "";  // Simple but not ideal

// Better: Remove each child
while (container.firstChild) {
    container.removeChild(container.firstChild);
}

// Even better: replaceChildren (modern)
container.replaceChildren();  // Removes all children

// Remove with condition
const items = document.querySelectorAll(".item");
items.forEach(item => {
    if (item.classList.contains("expired")) {
        item.remove();
    }
});

Cloning Elements

Copy elements with all their content:

const original = document.querySelector(".card");

// Clone element (shallow - no children)
const shallowClone = original.cloneNode();

// Clone element and all children (deep)
const deepClone = original.cloneNode(true);

// Add clone to document
document.querySelector(".container").appendChild(deepClone);

// Clone and modify
const template = document.querySelector(".template");
const newItem = template.cloneNode(true);
newItem.classList.remove("template");
newItem.classList.add("active");
newItem.querySelector(".title").textContent = "New Title";
document.querySelector(".list").appendChild(newItem);

Element Properties and Dimensions

Get information about elements:

const element = document.querySelector(".box");

// Basic properties
console.log(element.id);           // ID attribute
console.log(element.tagName);      // Tag name (uppercase: "DIV")
console.log(element.className);    // All classes as string

// Dimensions (including content only)
console.log(element.offsetWidth);   // Width including padding, border
console.log(element.offsetHeight);  // Height including padding, border
console.log(element.clientWidth);   // Width including padding (no border)
console.log(element.clientHeight);  // Height including padding (no border)

// Scroll dimensions
console.log(element.scrollWidth);   // Full scrollable width
console.log(element.scrollHeight);  // Full scrollable height
console.log(element.scrollTop);     // Current scroll position
console.log(element.scrollLeft);    // Current horizontal scroll

// Position
console.log(element.offsetLeft);    // Distance from left of offsetParent
console.log(element.offsetTop);     // Distance from top of offsetParent

// Position relative to viewport
const rect = element.getBoundingClientRect();
console.log(rect.top, rect.right, rect.bottom, rect.left);
console.log(rect.width, rect.height);

// Check if element is in viewport
function isInViewport(element) {
    const rect = element.getBoundingClientRect();
    return (
        rect.top >= 0 &&
        rect.left >= 0 &&
        rect.bottom <= window.innerHeight &&
        rect.right <= window.innerWidth
    );
}

Form Handling

Work with form elements:

const form = document.querySelector("form");
const input = document.querySelector("input[name='email']");

// Get/set input values
console.log(input.value);           // Current value
input.value = "new@example.com";    // Set value

// Check input types
const checkbox = document.querySelector("input[type='checkbox']");
console.log(checkbox.checked);      // true/false

const select = document.querySelector("select");

// Form submission
form.addEventListener("submit", (event) => {
    event.preventDefault();         // Stop page refresh
    
    // Get all form data
    const formData = new FormData(form);
    const data = Object.fromEntries(formData);
    console.log(data);
    
    // Or manually
    const email = form.querySelector("input[name='email']").value;
    const password = form.querySelector("input[name='password']").value;
    
    // Validate and submit
    if (validateForm(email, password)) {
        submitForm(data);
    }
});

// Real-time validation
input.addEventListener("input", (event) => {
    const value = event.target.value;
    if (value.length < 3) {
        event.target.setCustomValidity("At least 3 characters");
    } else {
        event.target.setCustomValidity("");
    }
});

Practical Examples

// Example 1: Dynamic todo list
function createTodoList() {
    const input = document.querySelector("#todo-input");
    const list = document.querySelector("#todo-list");
    const button = document.querySelector("#add-todo");
    
    button.addEventListener("click", () => {
        const text = input.value.trim();
        if (!text) return;
        
        const li = document.createElement("li");
        li.innerHTML = `
            ${text}
            
        `;
        
        li.querySelector(".delete").addEventListener("click", () => {
            li.remove();
        });
        
        list.appendChild(li);
        input.value = "";
    });
}

// Example 2: Modal window
function createModal(content) {
    const modal = document.createElement("div");
    modal.className = "modal-overlay";
    modal.innerHTML = `
        

            
            
${content}

        

    `;
    
    document.body.appendChild(modal);
    
    const closeModal = () => modal.remove();
    modal.querySelector(".modal-close").addEventListener("click", closeModal);
    modal.addEventListener("click", (e) => {
        if (e.target === modal) closeModal();
    });
    
    return modal;
}

// Example 3: Infinite scroll
function setupInfiniteScroll(callback) {
    const sentinel = document.querySelector("#sentinel");
    
    const observer = new IntersectionObserver((entries) => {
        if (entries[0].isIntersecting) {
            callback();
        }
    });
    
    observer.observe(sentinel);
}

Practice Exercise

// Exercise: Build a dynamic product list
const products = [
    { id: 1, name: "Laptop", price: 999, category: "Electronics" },
    { id: 2, name: "Phone", price: 699, category: "Electronics" },
    { id: 3, name: "Shirt", price: 29, category: "Clothing" },
    { id: 4, name: "Book", price: 15, category: "Books" }
];

// Create product card
function createProductCard(product) {
    const card = document.createElement("div");
    card.className = "product-card";
    card.dataset.id = product.id;
    
    card.innerHTML = `
        
${product.name}

        
$${product.price}

        
${product.category}

        
    `;
    
    return card;
}

// Render all products
function renderProducts(products) {
    const container = document.querySelector("#products");
    container.innerHTML = "";
    
    products.forEach(product => {
        const card = createProductCard(product);
        container.appendChild(card);
    });
}

// Filter by category
function filterProducts(category) {
    if (category === "all") return products;
    return products.filter(p => p.category === category);
}

// Sort products
function sortProducts(products, sortBy) {
    const sorted = [...products];
    switch(sortBy) {
        case "price-asc":
            return sorted.sort((a, b) => a.price - b.price);
        case "price-desc":
            return sorted.sort((a, b) => b.price - a.price);
        case "name":
            return sorted.sort((a, b) => a.name.localeCompare(b.name));
        default:
            return sorted;
    }
}

// Initialize
renderProducts(products);

// Event listeners for filters
document.querySelector("#category-filter").addEventListener("change", (e) => {
    const filtered = filterProducts(e.target.value);
    const sortBy = document.querySelector("#sort-filter").value;
    renderProducts(sortProducts(filtered, sortBy));
});

document.querySelector("#sort-filter").addEventListener("change", (e) => {
    const category = document.querySelector("#category-filter").value;
    const filtered = filterProducts(category);
    renderProducts(sortProducts(filtered, e.target.value));
});
