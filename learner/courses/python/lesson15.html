Why Best Practices Matter

Writing clean, maintainable Python code is crucial for professional development. Good practices make your code easier to read, debug, and scale. They help teams collaborate effectively and reduce technical debt. Following the Python community's conventions ensures your code is accessible to other developers.

PEP 8: Python Style Guide

PEP 8 is the official Python style guide. Following it makes your code consistent with the broader Python community.

Writing Clean Functions

Functions should do one thing, do it well, and have clear names. This makes code testable and reusable.

Function Best Practices:

# Good: Single responsibility
def calculate_discount(price, discount_rate):
    """Calculate the discounted price."""
    return price * (1 - discount_rate)

def format_price(price):
    """Format price as currency string."""
    return f"${price:.2f}"

# Bad: Multiple responsibilities
def calculate_and_print_discount(price, discount_rate):
    total = price * (1 - discount_rate)
    print(f"Discounted price: ${total:.2f}")
    # Mixing calculation and output is bad

# Good: Type hints for clarity
def get_user_age(user_id: int) -> int:
    """Get user age by ID.
    
    Args:
        user_id: The user's unique identifier
        
    Returns:
        The user's age in years
        
    Raises:
        UserNotFoundError: If user doesn't exist
    """
    user = database.get_user(user_id)
    return user.age

# Good: Default arguments for flexibility
def create_user(
    name: str,
    email: str,
    role: str = "user",
    active: bool = True
) -> dict:
    return {
        "name": name,
        "email": email,
        "role": role,
        "active": active
    }

# Avoid mutable default arguments!
# Bad - the list is shared across all calls
def add_item(item, items=[]):
    items.append(item)
    return items

# Good - use None and create inside function
def add_item(item, items=None):
    if items is None:
        items = []
    items.append(item)
    return items

# Keep functions small (under 20-30 lines)
# Extract complex logic into helper functions

Lines 19-32: Use type hints and thorough docstrings.

Lines 35-47: Default arguments make functions more flexible.

Lines 50-59: Never use mutable default arguments - use None instead.

Error Handling Patterns

Good error handling makes your code robust and helps with debugging. Follow these patterns for professional code.

Error Handling Best Practices:

# Be specific about exceptions
# Bad
try:
    do_something()
except Exception:  # Too broad
    pass

# Good
try:
    do_something()
except (ValueError, KeyError) as e:
    logger.error(f"Operation failed: {e}")
    raise

# Use custom exceptions for domain errors
class PaymentError(Exception):
    """Base exception for payment errors."""
    pass

class InsufficientFundsError(PaymentError):
    """Account has insufficient funds."""
    def __init__(self, balance, amount):
        self.balance = balance
        self.amount = amount
        super().__init__(
            f"Insufficient funds: {balance} available, "
            f"{amount} required"
        )

# Handle at the right level
def process_payment(user_id, amount):
    # Low-level: let exceptions propagate
    user = get_user(user_id)
    if user.balance < amount:
        raise InsufficientFundsError(user.balance, amount)
    return charge_user(user, amount)

def checkout(user_id, cart_total):
    # High-level: handle and present to user
    try:
        receipt = process_payment(user_id, cart_total)
        return {"success": True, "receipt": receipt}
    except InsufficientFundsError as e:
        return {
            "success": False,
            "error": "insufficient_funds",
            "message": str(e)
        }
    except PaymentError as e:
        logger.error(f"Payment failed: {e}")
        return {"success": False, "error": "payment_failed"}

# Use context managers for cleanup
from contextlib import contextmanager

@contextmanager
def database_transaction():
    conn = get_connection()
    try:
        yield conn
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        conn.close()

# Usage
with database_transaction() as conn:
    conn.execute("INSERT INTO users VALUES (?)", (user,))

"""
Module-level docstring.
Describe what this module does.

class UserService:
    """Service for managing user operations.
    
    This class provides methods for creating, updating,
    and deleting users in the system.
    
    Attributes:
        db: Database connection instance
        cache: Optional cache for user data
    
    Example:
        >>> service = UserService(database)
        >>> user = service.get_user(1)
        >>> print(user.name)
        'Alice'
    """
    
    def __init__(self, db, cache=None):
        """Initialize the user service.
        
        Args:
            db: Database connection
            cache: Optional cache backend (default: None)
        """
        self.db = db
        self.cache = cache
    
    def get_user(self, user_id: int) -> dict:
        """Retrieve a user by ID.
        
        Args:
            user_id: The unique identifier for the user
        
        Returns:
            Dictionary containing user data
        
        Raises:
            UserNotFoundError: If user with given ID doesn't exist
            DatabaseError: If database operation fails
        
        Example:
            >>> user = service.get_user(1)
            >>> user['name']
            'Alice'
        """
        if self.cache:
            cached = self.cache.get(f"user:{user_id}")
            if cached:
                return cached
        
        user = self.db.query(
            "SELECT * FROM users WHERE id = ?",
            (user_id,)
        )
        
        if not user:
            raise UserNotFoundError(user_id)
        
        return user

# models.py
"""Data models for the task manager."""
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional
from enum import Enum

class Priority(Enum):
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"

class Status(Enum):
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"

@dataclass
class Task:
    """Represents a task in the system."""
    title: str
    description: str = ""
    priority: Priority = Priority.MEDIUM
    status: Status = Status.PENDING
    id: Optional[int] = None
    created_at: datetime = field(default_factory=datetime.now)
    completed_at: Optional[datetime] = None
    
    def mark_complete(self):
        """Mark the task as completed."""
        self.status = Status.COMPLETED
        self.completed_at = datetime.now()
    
    def __str__(self):
        status_icon = "✓" if self.status == Status.COMPLETED else "○"
        return f"[{status_icon}] {self.id}. {self.title} ({self.priority.value})"

Storage and Manager:

# storage.py
"""Persistent storage for tasks."""
import json
from pathlib import Path
from typing import List, Optional
from .models import Task, Priority, Status

class TaskStorage:
    """Handle persistent storage of tasks."""
    
    def __init__(self, filepath: str = "tasks.json"):
        self.filepath = Path(filepath)
    
    def load_all(self) -> List[Task]:
        """Load all tasks from storage."""
        if not self.filepath.exists():
            return []
        
        with open(self.filepath) as f:
            data = json.load(f)
        
        return [self._deserialize(item) for item in data]
    
    def save_all(self, tasks: List[Task]):
        """Save all tasks to storage."""
        data = [self._serialize(task) for task in tasks]
        with open(self.filepath, "w") as f:
            json.dump(data, f, indent=2, default=str)
    
    def _serialize(self, task: Task) -> dict:
        """Convert Task to dictionary."""
        return {
            "id": task.id,
            "title": task.title,
            "description": task.description,
            "priority": task.priority.value,
            "status": task.status.value,
            "created_at": task.created_at.isoformat(),
            "completed_at": task.completed_at.isoformat() if task.completed_at else None
        }
    
    def _deserialize(self, data: dict) -> Task:
        """Convert dictionary to Task."""
        return Task(
            id=data["id"],
            title=data["title"],
            description=data["description"],
            priority=Priority(data["priority"]),
            status=Status(data["status"]),
            created_at=datetime.fromisoformat(data["created_at"]),
            completed_at=datetime.fromisoformat(data["completed_at"]) if data["completed_at"] else None
        )

# manager.py
"""Task management logic."""
from typing import List, Optional
from .models import Task, Priority, Status
from .storage import TaskStorage

class TaskManager:
    """Manage task operations."""
    
    def __init__(self, storage: TaskStorage):
        self.storage = storage
        self._tasks: List[Task] = []
        self._load_tasks()
    
    def _load_tasks(self):
        """Load tasks from storage."""
        self._tasks = self.storage.load_all()
    
    def _save_tasks(self):
        """Save tasks to storage."""
        self.storage.save_all(self._tasks)
    
    def _next_id(self) -> int:
        """Get the next available ID."""
        if not self._tasks:
            return 1
        return max(t.id for t in self._tasks if t.id) + 1
    
    def add_task(
        self,
        title: str,
        description: str = "",
        priority: Priority = Priority.MEDIUM
    ) -> Task:
        """Add a new task."""
        task = Task(
            id=self._next_id(),
            title=title,
            description=description,
            priority=priority
        )
        self._tasks.append(task)
        self._save_tasks()
        return task
    
    def get_task(self, task_id: int) -> Optional[Task]:
        """Get a task by ID."""
        for task in self._tasks:
            if task.id == task_id:
                return task
        return None
    
    def complete_task(self, task_id: int) -> Optional[Task]:
        """Mark a task as completed."""
        task = self.get_task(task_id)
        if task:
            task.mark_complete()
            self._save_tasks()
        return task
    
    def delete_task(self, task_id: int) -> bool:
        """Delete a task by ID."""
        task = self.get_task(task_id)
        if task:
            self._tasks.remove(task)
            self._save_tasks()
            return True
        return False
    
    def list_tasks(
        self,
        status: Optional[Status] = None,
        priority: Optional[Priority] = None
    ) -> List[Task]:
        """List tasks with optional filters."""
        tasks = self._tasks
        if status:
            tasks = [t for t in tasks if t.status == status]
        if priority:
            tasks = [t for t in tasks if t.priority == priority]
        return sorted(tasks, key=lambda t: t.id)

CLI Interface:

# cli.py
"""Command-line interface for task manager."""
import argparse
import sys
from .manager import TaskManager
from .storage import TaskStorage
from .models import Priority, Status

class TaskCLI:
    """Command-line interface for task management."""
    
    def __init__(self):
        storage = TaskStorage()
        self.manager = TaskManager(storage)
    
    def run(self, args=None):
        """Run the CLI."""
        parser = self._create_parser()
        parsed = parser.parse_args(args)
        
        if parsed.command is None:
            parser.print_help()
            return
        
        command = getattr(self, f"cmd_{parsed.command}")
        command(parsed)
    
    def _create_parser(self):
        """Create argument parser."""
        parser = argparse.ArgumentParser(
            description="Task Manager CLI"
        )
        subparsers = parser.add_subparsers(dest="command")
        
        # Add command
        add_parser = subparsers.add_parser("add", help="Add a task")
        add_parser.add_argument("title", help="Task title")
        add_parser.add_argument("-d", "--description", default="")
        add_parser.add_argument("-p", "--priority", 
            choices=["low", "medium", "high"], default="medium")
        
        # List command
        list_parser = subparsers.add_parser("list", help="List tasks")
        list_parser.add_argument("-s", "--status",
            choices=["pending", "in_progress", "completed"])
        list_parser.add_argument("-p", "--priority",
            choices=["low", "medium", "high"])
        
        # Complete command
        complete_parser = subparsers.add_parser("complete", help="Complete task")
        complete_parser.add_argument("id", type=int, help="Task ID")
        
        # Delete command
        delete_parser = subparsers.add_parser("delete", help="Delete task")
        delete_parser.add_argument("id", type=int, help="Task ID")
        
        return parser
    
    def cmd_add(self, args):
        """Add a new task."""
        priority = Priority(args.priority)
        task = self.manager.add_task(
            title=args.title,
            description=args.description,
            priority=priority
        )
        print(f"Added task {task.id}: {task.title}")
    
    def cmd_list(self, args):
        """List tasks."""
        status = Status(args.status) if args.status else None
        priority = Priority(args.priority) if args.priority else None
        
        tasks = self.manager.list_tasks(status, priority)
        
        if not tasks:
            print("No tasks found.")
            return
        
        for task in tasks:
            print(task)
    
    def cmd_complete(self, args):
        """Complete a task."""
        task = self.manager.complete_task(args.id)
        if task:
            print(f"Completed task {task.id}: {task.title}")
        else:
            print(f"Task {args.id} not found.")
    
    def cmd_delete(self, args):
        """Delete a task."""
        if self.manager.delete_task(args.id):
            print(f"Deleted task {args.id}")
        else:
            print(f"Task {args.id} not found.")

def main():
    """Entry point for the CLI."""
    cli = TaskCLI()
    cli.run()

if __name__ == "__main__":
    main()
